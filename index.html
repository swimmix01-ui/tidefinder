async function fetchMarineWeather() { const today = new Date(); const y = today.getFullYear(); const m = String(today.getMonth() + 1).padStart(2, '0'); const d = String(today.getDate()).padStart(2, '0'); const reqDate = `${y}${m}${d}`; try { const [waterRes, airRes, pressRes, fogRes] = await Promise.all([ fetch(`/.netlify/functions/weather?mode=watertemp&reqDate=${reqDate}`), fetch(`/.netlify/functions/weather?mode=airtemp&reqDate=${reqDate}`), fetch(`/.netlify/functions/weather?mode=airpress&reqDate=${reqDate}`), fetch(`/.netlify/functions/weather?mode=seafog&obsCode=DT_0091&reqDate=${reqDate}`) // ◀ 포항 관측소 고정 ]); const waterData = await waterRes.json(); const airData = await airRes.json(); const pressData = await pressRes.json(); // 시정 데이터는 에러가 나거나 비어 있어도 전체 앱이 멈추지 않도록 안전하게 처리 let fogData = null; try { fogData = await fogRes.json(); } catch(e) { console.warn("시정 데이터 파싱 실패 (정비 중 또는 빈 데이터):", e); } const waterItems = waterData?.body?.items?.item || waterData?.response?.body?.items?.item; const airItems = airData?.body?.items?.item || airData?.response?.body?.items?.item; const pressItems = pressData?.body?.items?.item || pressData?.response?.body?.items?.item; const fogItems = fogData?.body?.items?.item || fogData?.response?.body?.items?.item; // 1) 수온 if (waterItems && waterItems.length > 0) { const latestWater = waterItems[waterItems.length - 1]; document.getElementById('marineWaterTemp').textContent = `${latestWater.wtem ?? latestWater.value ?? '-'}°C`; document.getElementById('marineObsTime').textContent = latestWater.obsrvnDt ?? today.toLocaleTimeString(); } // 2) 기온 if (airItems && airItems.length > 0) { const latestAir = airItems[airItems.length - 1]; const airVal = latestAir.artmp ?? latestAir.atemp ?? latestAir.temp ?? latestAir.value; document.getElementById('marineAirTemp').textContent = (airVal !== undefined) ? `${airVal}°C` : '확인필요'; } // 3) 기압 if (pressItems && pressItems.length > 0) { const latestPress = pressItems[pressItems.length - 1]; const pressVal = latestPress.atmpr ?? latestPress.press ?? latestPress.airPress ?? latestPress.value; document.getElementById('marineAirPress').textContent = (pressVal !== undefined) ? `${pressVal}hPa` : '확인필요'; } // 4) 시정 (해무) 안전 파싱 및 강제 출력 로직 // 국립해양조사원 장비 특성상 안개가 없고 맑으면 데이터 배열이 아예 비어서 옵니다. if (fogItems && fogItems.length > 0) { const latestFog = fogItems[fogItems.length - 1]; // 해양조사원 표준 실측 필드명 후보군 전수 검사 (visb, value, vis) const visbVal = latestFog.visb ?? latestFog.value ?? latestFog.vis; if (visbVal !== undefined && visbVal !== null && visbVal !== "") { const meters = parseFloat(visbVal); // m 단위를 실무에서 보기 편한 km 단위로 치환 const displayVisb = meters >= 1000 ? `${(meters / 1000).toFixed(1)}km` : `${meters}m`; document.getElementById('marineVisibility').textContent = displayVisb; } else { document.getElementById('marineVisibility').textContent = '양호 (10km+)'; } } else { // 데이터가 비어있다는 것은 안개가 전혀 없는 최상의 가시거리 상태를 의미하는 경우가 대부분입니다. document.getElementById('marineVisibility').textContent = '양호 (10km+)'; } } catch (err) { console.error('해양 날씨 연동 에러:', err); // 시정 칸이 빈 채로 멈추지 않도록 에러 발생 시 최하단 방어선 구축 if(!document.getElementById('marineVisibility').textContent || document.getElementById('marineVisibility').textContent === '-') { document.getElementById('marineVisibility').textContent = '확인필요'; } } }
